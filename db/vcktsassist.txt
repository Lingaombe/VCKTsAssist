create table Users(
subjectID varchar(100) not null,
id int primary key auto_increment,
username varchar(100) not null,
upassword varchar(100) not null,
urole varchar(100) not null,
email varchar(100) not null unique,
foreign key (subjectID) references Subjects(subjectID)
);

create table Streams(
streamID varchar(100) primary key,
streamName varchar(100) not null,
streamLevel varchar(100) not null
);

create table Subjects(
streamID varchar(100) not null,
subjectID varchar(100) primary key,
subjectName varchar(100) not null,
foreign key (streamID) references Streams(streamID)
);

create table Courses(
subjectID varchar(100) not null,
courseID varchar(100) primary key,
courseSem varchar(100) not null,
courseName varchar(100) not null,
marksInternal int,
marksExternal int,
marksPractical int,
foreign key (subjectID) references Subjects(subjectID)
);


create table questionBanks(
courseID varchar(100) not null,
questionBankID int primary key auto_increment,
questionBankType varchar(100) not null,
questionBankName varchar(100) unique not null,
foreign key (courseID) references Courses(courseID)
);


create table questions(
questionBankID int not null,
questionID int primary key auto_increment,
questionBody varchar(1000) unique not null,
questionGrade varchar(100) not null,
questionPhoto varchar(200),
questionUnit varchar(100) not null,
questionOption1 varchar(100),
questionOption2 varchar(100),
questionOption3 varchar(100),
questionOption4 varchar(100),
questionMarks int not null,
questionUsed boolean default false,
foreign key (questionBankID) references questionBanks(questionBankID)
);

create table Teachers(
courseID varchar(100),
teacherID int,
foreign key (courseID) references Courses(courseID),
foreign key (teacherID) references Users(id)
);
